const int maxn=1e6;
int par[maxn],rnk[maxn];
int find(int a)
{
	return a == par[a] ? a : par[a] = find(par[a]);
}
void merge(int a, int b)
{
	a = find(a), b = find(b);
	if(rnk[a] > rnk[b])
		swap(a, b);
	par[a] = b;
	rnk[b] += rnk[a];
}
 
bool same(int a, int b) {
    return find(a) == find(b);
}
